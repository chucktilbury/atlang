163  if(config == NULL) {

175  }
176  }

179  case CONFIG_TYPE_NUM: {
180  char* value;

183  else {
184  idx++;
185  value = argv[idx];
186  }
187

189  if(num == 0 && errno != 0) {

191  show_use();
192  }
193  config->value.number = num;
194  config->touched++;
195  }
196  break;

292 char* iterate_config(const char* name) {
293 
294  char* retv = NULL;
295  configuration_t* config = find_config_by_name(name);
296  if(config == NULL) {
297  fprintf(stderr, "CFG ERROR: Cannot find configuration parameter: \"%s\"\n", name);
298  exit(1);
299  }
300 
301  switch(config->type) {
302  case CONFIG_TYPE_NUM:
303  case CONFIG_TYPE_BOOL:
304  // there can be only exactly one value here. Allocate the buffer if it does not exist
305  // and then return it.
306  if(config->iter_buf == NULL) {
307  config->iter_buf = CALLOC(12, sizeof(char)); // maximum number of characters in an int.
308  snprintf(config->iter_buf, 12, "%d", config->value.number);
309  }
310  else {
311  FREE(config->iter_buf);
312  config->iter_buf = NULL;
313  }
314  retv = config->iter_buf;
315  break;
316 
317  case CONFIG_TYPE_STR:
318  // there can be only exactly one value here. It's already a (char*).
319  if(config->iter_buf == NULL)
320  config->iter_buf = config->value.string;
321  else
322  config->iter_buf = NULL;
323  retv = config->iter_buf;
324  break;
325 
326  case CONFIG_TYPE_LIST:
327  // This type actually gets iterated. use strtok_r() to iterate it.
328  if(config->iter_buf != NULL) {
329  config->iter_buf = get_ptr_list_next(config->value.list);
330  retv = config->iter_buf;
331  }
332  else {
333  reset_ptr_list(config->value.list);
334  config->iter_buf = get_ptr_list_next(config->value.list);
335  retv = config->iter_buf;
336  }
337  break;
338 
339  default:
340  fprintf(stderr, "CFG ERROR: Cannot iterate configuation parameter\n");
341  exit(1);
342  }
343  return retv;
344 }
